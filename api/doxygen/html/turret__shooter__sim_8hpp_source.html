93 DoubleDifferentialMechanism::InverseResult solveMotorStateForAim(
94 double yaw_rad, double pitch_rad, double yaw_velocity_radps = 0.0,
95 double pitch_velocity_radps = 0.0) const {
96 DoubleDifferentialMechanism::JointState desired{};
97 desired.yaw_rad = yaw_rad;

112 bool applyAim(double yaw_rad, double pitch_rad,
113 double yaw_velocity_radps = 0.0,
114 double pitch_velocity_radps = 0.0) {
115 const auto solved = solveMotorStateForAim(
116 yaw_rad, pitch_rad, yaw_velocity_radps, pitch_velocity_radps);
117 if (!solved.valid) {
118 return false;
119 }
120 motor_state_ = solved.motor_state;
121 return true;
122 }

126 DoubleDifferentialMechanism::JointState jointState() const {

158 bool pickupBallNearby() {
159 BallPhysicsSim3D::PickupRequest request{};

162 request.carry_offset_m =

172 void shoot(double muzzle_speed_mps,
173 const Vector3& spin_radps = Vector3::zero()) {
174 const Vector3 muzzle_position = muzzlePositionWorld();
175 const Vector3 shot_velocity =
176 base_velocity_mps_ +
177 muzzleDirectionWorld() * std::max(0.0, muzzle_speed_mps);
178 ball_.shoot(muzzle_position, shot_velocity, spin_radps);
179 }

189 }

195 void step(double dt_s) {
196 const Vector3 carry_position = intakePositionWorld();

206 private:
207 Quaternion turretOrientationWorld() const {
208 const auto joints = jointState();
209 const Quaternion yaw =
210 Quaternion::fromAxisAngle(Vector3::unitZ(), joints.yaw_rad);
211 const Quaternion pitch =
212 Quaternion::fromAxisAngle(Vector3::unitY(), joints.pitch_rad);
213 return yaw * pitch;
214 }
