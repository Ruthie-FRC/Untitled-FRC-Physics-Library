532 std::size_t robot_index, const ExitTrajectoryParameters& command,
533 bool spawn_on_ground_after_touch = true,
534 const std::function<void()>& hit_target_callback = {}) {
535 std::lock_guard<std::mutex> lock(state_mutex_);
536 if (robot_index >= robots_.size()) {
537 return kNoBall;
538 }
539
540 const RobotState& robot = robots_[robot_index];
541 const double launch_speed =
542 command.estimated_exit_velocity_mps > 0.0
543 ? command.estimated_exit_velocity_mps
544 : std::max(0.0, command.mechanism_speed_mps);
545
546 const double yaw_world = robot.yaw_rad + command.yaw_offset_rad;
547 const double cos_pitch = std::cos(command.pitch_rad);
548 const Vector3 direction_world(std::cos(yaw_world) * cos_pitch,
549 std::sin(yaw_world) * cos_pitch,
550 std::sin(command.pitch_rad));
551
552 ProjectileEntity projectile{};
553 projectile.type = command.gamepiece_type;
554 projectile.position_m =
555 robot.position_m + rotateYaw(command.launch_offset_m, robot.yaw_rad);
556 projectile.velocity_mps =
557 robot.velocity_mps + direction_world * launch_speed;
558 projectile.spin_radps = command.spin_radps;
559 projectile.gravity_mps2 =
560 std::max(0.0, -fallbackBallConfig().gravity_mps2.z);
561 projectile.spawn_on_ground_after_touch = spawn_on_ground_after_touch;
562 projectile.hit_target_callback = hit_target_callback;
563
564 projectiles_.push_back(projectile);
565 return projectiles_.size() - 1;

610 void stepSingle(double dt_s) {
611 if (dt_s <= 0.0) {
612 return;
613 }
614
615 integrateRobots(dt_s);
616 resolveRobotRobotImpedance();
617 updateProjectiles(dt_s);
618
619 for (std::size_t i = 0; i < balls_.size(); ++i) {
620 BallEntity& ball = balls_[i];
621
622 const Vector3 pre_step_position = ball.sim.state().position_m;
623 const bool skip_integration =
624 field_.sleeping_enabled && ball.sleeping;
625 if (!skip_integration) {
626 ball.sim.step(dt_s);
627 }
628
629 auto state = ball.sim.state();
630 const double spin_decay =
631 std::max(0.0, 1.0 - field_.free_ball_spin_decay_per_s * dt_s);
632 state.spin_radps *= spin_decay;
633 ball.sim.setState(state);
634
635 resolveContinuousCollision(ball, pre_step_position, dt_s);
636 resolveRobotBallContacts(ball);
637 resolveFieldElements(ball, dt_s);
638 resolveFieldBounds(ball);
639 }
640
641 resolveBallBallContacts(dt_s);
642 updateSleepStates();
643 }

653 void updateProjectiles(double dt_s) {
654 const double floor_z = fallbackBallConfig().ground_height_m;
655
656 for (auto& projectile : projectiles_) {
657 if (!projectile.active) {
658 continue;
659 }
660
661 projectile.age_s += dt_s;
662 projectile.velocity_mps.z -= projectile.gravity_mps2 * dt_s;
663 projectile.position_m += projectile.velocity_mps * dt_s;
664
665 if (checkProjectileGoalHit(projectile)) {
666 projectile.hit_target = true;
667 projectile.active = false;
668 if (projectile.hit_target_callback) {
669 projectile.hit_target_callback();
670 }
671 continue;
672 }
673
674 if (projectile.position_m.z <= floor_z) {
675 if (projectile.spawn_on_ground_after_touch) {
676 const GamePieceInfo* info = findGamePieceInfo(projectile.type);
677 const auto& physics_cfg =
678 (info != nullptr) ? info->physics_config : fallbackBallConfig();
679 const auto& physics_props = (info != nullptr)
680 ? info->ball_properties
681 : fallbackBallProperties();
682
683 BallPhysicsSim3D::BallState state{};
684 state.position_m = Vector3(
685 projectile.position_m.x, projectile.position_m.y,
686 std::max(physics_props.radius_m, physics_cfg.ground_height_m));
687 state.velocity_mps = projectile.velocity_mps;
688 state.spin_radps = projectile.spin_radps;
689 addBall(state, physics_cfg, physics_props);
690 if (!ball_types_.empty()) {
691 ball_types_.back() = projectile.type;
692 }
693 }
694 projectile.active = false;
695 continue;
696 }
697
698 if (isProjectileOutOfField(projectile)) {
699 projectile.active = false;
700 }
701 }

704 bool checkProjectileGoalHit(const ProjectileEntity& projectile) const {
705 for (const auto& goal : goals_) {
706 if (!goal.accept_any_type && goal.accepted_type != projectile.type) {
707 continue;
708 }
709 if (goal.require_positive_vertical_velocity &&
710 projectile.velocity_mps.z <= 0.0) {
711 continue;
712 }
713 if (goal.custom_velocity_validator &&
714 !goal.custom_velocity_validator(projectile.velocity_mps)) {
715 continue;
716 }
717
718 if (goal.shape == GoalZone::Shape::kBox) {
719 const Vector3 delta = projectile.position_m - goal.center_m;
720 if (std::abs(delta.x) <= goal.half_extents_m.x &&
721 std::abs(delta.y) <= goal.half_extents_m.y &&
722 std::abs(delta.z) <= goal.half_extents_m.z) {
723 return true;
724 }
725 } else {
726 if ((projectile.position_m - goal.center_m).norm() <= goal.radius_m) {
727 return true;
728 }
729 }
730 }
731 return false;
732 }

844 static void applyContactImpulse(Vector3& velocity, const Vector3& normal,
845 double restitution, double friction) {
846 const double vn = velocity.dot(normal);
847 if (vn >= 0.0) {
848 return;
849 }
850
851 velocity -= normal * ((1.0 + std::clamp(restitution, 0.0, 1.0)) * vn);
852 const Vector3 tangential = velocity - normal * velocity.dot(normal);
853 velocity -= tangential * std::clamp(friction, 0.0, 1.0);
854 }

860 static double velocityScaledRestitution(
861 double base_restitution, double impact_speed_mps,
862 double reference_speed_mps, double speed_exponent,
863 double min_scale) {
864 const double base = std::clamp(base_restitution, 0.0, 1.0);
865 if (!std::isfinite(impact_speed_mps) || impact_speed_mps <= 0.0) {
866 return base;
867 }
868
869 const double reference_speed = std::max(
870 1e-6,
871 (std::isfinite(reference_speed_mps) && reference_speed_mps >= 0.0)
872 ? reference_speed_mps
873 : 3.0);
874 const double exponent =
875 (std::isfinite(speed_exponent) && speed_exponent >= 0.0)
876 ? speed_exponent
877 : 0.15;
878 const double clamped_min_scale = std::clamp(
879 std::isfinite(min_scale) ? min_scale : 0.55, 0.0, 1.0);
880
881 if (impact_speed_mps <= reference_speed || exponent <= 0.0) {
882 return base;
883 }
884
885 const double speed_scale =
886 std::pow(reference_speed / impact_speed_mps, exponent);
887 const double restitution_scale =
888 std::clamp(speed_scale, clamped_min_scale, 1.0);
889 return base * restitution_scale;
890 }

899 void updateSleepStates() {
900 if (!field_.sleeping_enabled) {
901 for (auto& ball : balls_) {
902 wakeBall(ball);
903 }
904 return;
905 }
906
907 const double velocity_threshold = std::max(0.0, field_.sleep_velocity_threshold_mps);
908 const double spin_threshold = std::max(0.0, field_.sleep_spin_threshold_radps);
909 const int frame_threshold = std::max(1, field_.sleep_frame_threshold);
910
911 for (auto& ball : balls_) {
912 auto state = ball.sim.state();
913 if (state.held || ball.scored_in_net) {
914 wakeBall(ball);
915 continue;
916 }
917
918 const double floor_z =
919 ball.sim.config().ground_height_m + ball.sim.ballProperties().radius_m;
920 const bool near_ground = state.position_m.z <= floor_z + 0.01;
921 const bool sleep_candidate = near_ground &&
922 state.velocity_mps.norm() <= velocity_threshold &&
923 state.spin_radps.norm() <= spin_threshold;
924
925 if (!sleep_candidate) {
926 wakeBall(ball);
927 continue;
928 }
929
930 ball.sleep_frame_count += 1;
931 if (ball.sleep_frame_count < frame_threshold) {
932 continue;
933 }
934
935 ball.sleeping = true;
936 state.velocity_mps = Vector3::zero();
937 state.spin_radps = Vector3::zero();
938 ball.sim.setState(state);
939 }

948 void resolveContinuousCollision(BallEntity& ball,
949 const Vector3& previous_position,
950 double dt_s) const {
951 if (!field_.ccd_enabled || dt_s <= 0.0 || ball.sim.state().held) {
952 return;
953 }
954
955 auto state = ball.sim.state();
956 const double speed_mps =
957 (state.position_m - previous_position).norm() / std::max(1e-9, dt_s);
958 if (speed_mps < std::max(0.0, field_.ccd_speed_threshold_mps)) {
959 return;
960 }
961
962 const double radius = ball.sim.ballProperties().radius_m;
963 resolveSweptFieldBounds(state, previous_position, radius);
964
965 for (const auto& boundary : field_elements_) {
966 if (!boundary.is_active) {
967 continue;
968 }
969
970 switch (boundary.type) {
971 case BoundaryType::kPlane:
972 case BoundaryType::kWall:
973 resolveSweptPlaneBoundary(state, previous_position, boundary, radius);
974 break;
975 case BoundaryType::kBox:
976 resolveSweptBoxBoundary(state, previous_position, boundary, radius);
977 break;
978 default:
979 break;
980 }
981 }
982
983 ball.sim.setState(state);
984 }

987 void resolveSweptFieldBounds(BallPhysicsSim3D::BallState& state,
988 const Vector3& previous_position,

1057 void resolveSweptBoxBoundary(BallPhysicsSim3D::BallState& state,
1058 const Vector3& previous_position,
1059 const EnvironmentalBoundary& boundary,
1060 double ball_radius) const {
1061 const Quaternion inverse = boundary.orientation.inverse();
1062 const Vector3 prev_local =
1063 inverse.rotate(previous_position - boundary.position_m);
1064 const Vector3 curr_local =
1065 inverse.rotate(state.position_m - boundary.position_m);
1066 const Vector3 delta_local = curr_local - prev_local;
1067
1068 const Vector3 expanded_min(-boundary.half_extents_m.x - ball_radius,
1069 -boundary.half_extents_m.y - ball_radius,
1070 -boundary.half_extents_m.z - ball_radius);
1071 const Vector3 expanded_max(boundary.half_extents_m.x + ball_radius,
1072 boundary.half_extents_m.y + ball_radius,
1073 boundary.half_extents_m.z + ball_radius);
1074
1075 auto inside_expanded = [&](const Vector3& p) {
1076 return p.x >= expanded_min.x && p.x <= expanded_max.x &&
1077 p.y >= expanded_min.y && p.y <= expanded_max.y &&
1078 p.z >= expanded_min.z && p.z <= expanded_max.z;
1079 };
1080
1081 if (inside_expanded(curr_local)) {
1082 return;
1083 }
1084
1085 double t_enter = 0.0;
1086 double t_exit = 1.0;
1087 Vector3 local_normal = Vector3::zero();
1088
1089 auto process_axis = [&](double p0, double d, double min_b, double max_b,
1090 const Vector3& min_normal,
1091 const Vector3& max_normal) -> bool {
1092 if (std::abs(d) < 1e-9) {
1093 return p0 >= min_b && p0 <= max_b;
1094 }
1095
1096 const double inv_d = 1.0 / d;
1097 double t0 = (min_b - p0) * inv_d;
1098 double t1 = (max_b - p0) * inv_d;
1099 Vector3 candidate_normal = min_normal;
1100 if (t0 > t1) {
1101 std::swap(t0, t1);
1102 candidate_normal = max_normal;
1103 }
1104
1105 if (t0 > t_enter) {
1106 t_enter = t0;
1107 local_normal = candidate_normal;
1108 }
1109 t_exit = std::min(t_exit, t1);
1110 return t_enter <= t_exit;
1111 };
1112
1113 if (!process_axis(prev_local.x, delta_local.x, expanded_min.x, expanded_max.x,
1114 Vector3(-1.0, 0.0, 0.0), Vector3(1.0, 0.0, 0.0))) {
1115 return;
1116 }
1117 if (!process_axis(prev_local.y, delta_local.y, expanded_min.y, expanded_max.y,
1118 Vector3(0.0, -1.0, 0.0), Vector3(0.0, 1.0, 0.0))) {
1119 return;
1120 }
1121 if (!process_axis(prev_local.z, delta_local.z, expanded_min.z, expanded_max.z,
1122 Vector3(0.0, 0.0, -1.0), Vector3(0.0, 0.0, 1.0))) {
1123 return;
1124 }
1125
1126 if (t_enter < 0.0 || t_enter > 1.0 || local_normal.isZero()) {
1127 return;
1128 }
1129
1130 const Vector3 hit_local = prev_local + delta_local * t_enter;
1131 const Vector3 corrected_local = hit_local + local_normal * 1e-4;
1132 const Vector3 world_normal =
1133 boundary.orientation.rotate(local_normal).normalized();
1134
1135 state.position_m = boundary.position_m + boundary.orientation.rotate(corrected_local);
1136 const double impact_speed = std::max(0.0, -state.velocity_mps.dot(world_normal));
1137 applyContactImpulse(state.velocity_mps, world_normal,
1138 scaledWallRestitution(boundary.restitution,
1139 impact_speed),
1140 boundary.friction_coefficient);
1141 }

1202 for (std::size_t j = i + 1; j < robots_.size(); ++j) {
1203 RobotState& a = robots_[i];

1205
1206 Vector3 delta = b.position_m - a.position_m;
1207 delta.z = 0.0;

1211 continue;
1212 }
1213
1214 Vector3 normal = distance > 1e-9 ? delta / distance : Vector3::unitX();

1216 a.position_m -= normal * (0.5 * overlap);
1217 b.position_m += normal * (0.5 * overlap);
1218

1220 (a.velocity_mps - b.velocity_mps).dot(normal);

1222 continue;
1223 }
1224

1226 a.velocity_mps += normal * impulse;
1227 b.velocity_mps -= normal * impulse;

1276 void resolveBallBallContacts(double dt_s) {
1277 struct BallBallContact {
1278 std::size_t a_index{};
1279 std::size_t b_index{};
1280 Vector3 normal{};
1281 double penetration{0.0};
1282 double inv_mass_a{0.0};
1283 double inv_mass_b{0.0};
1284 double mass_a{0.0};
1285 double mass_b{0.0};
1286 double radius_a{0.0};
1287 double radius_b{0.0};
1288 double restitution{0.0};
1289 double friction{0.0};
1290 double normal_impulse_accum{0.0};
1291 double tangent_impulse_accum{0.0};
1292 };
1293
1294 std::vector<BallBallContact> contacts;
1295 contacts.reserve(balls_.size());
1296
1297 for (std::size_t i = 0; i < balls_.size(); ++i) {
1298 const auto& state_a = balls_[i].sim.state();
1299 if (state_a.held || balls_[i].scored_in_net) {
1300 continue;
1301 }
1302
1303 const double radius_a = balls_[i].sim.ballProperties().radius_m;
1304 const double mass_a =
1305 std::max(1e-9, balls_[i].sim.ballProperties().mass_kg);
1306
1307 for (std::size_t j = i + 1; j < balls_.size(); ++j) {
1308 const auto& state_b = balls_[j].sim.state();
1309 if (state_b.held || balls_[j].scored_in_net) {
1310 continue;
1311 }
1312
1313 const double radius_b = balls_[j].sim.ballProperties().radius_m;
1314 const double mass_b =
1315 std::max(1e-9, balls_[j].sim.ballProperties().mass_kg);
1316
1317 const Vector3 delta = state_b.position_m - state_a.position_m;
1318 const double distance = delta.norm();
1319 const double target_distance = radius_a + radius_b;
1320 if (distance >= target_distance) {
1321 continue;
1322 }
1323
1324 BallBallContact contact{};
1325 contact.a_index = i;
1326 contact.b_index = j;
1327 contact.normal = distance > 1e-9 ? delta / distance : Vector3::unitX();
1328 contact.penetration = target_distance - distance;
1329 contact.inv_mass_a = 1.0 / mass_a;
1330 contact.inv_mass_b = 1.0 / mass_b;
1331 contact.mass_a = mass_a;
1332 contact.mass_b = mass_b;
1333 contact.radius_a = radius_a;
1334 contact.radius_b = radius_b;
1335 const double approach_speed =
1336 std::max(0.0, (state_a.velocity_mps - state_b.velocity_mps).dot(contact.normal));
1337 contact.restitution = velocityScaledRestitution(
1338 field_.ball_ball_contact_restitution, approach_speed,
1339 field_.ball_ball_restitution_reference_speed_mps,
1340 field_.ball_ball_restitution_speed_exponent,
1341 field_.ball_ball_restitution_min_scale);
1342 contact.friction = std::clamp(field_.ball_ball_contact_friction, 0.0, 1.0);
1343 contacts.push_back(contact);
1344 wakeBall(balls_[i]);
1345 wakeBall(balls_[j]);
1346 }
1347 }
1348
1349 if (contacts.empty()) {
1350 return;
1351 }
1352
1353 const int solver_iterations = std::max(1, field_.solver_iterations);
1354 const double inv_dt = 1.0 / std::max(1e-9, dt_s);
1355
1356 for (int iteration = 0; iteration < solver_iterations; ++iteration) {
1357 for (auto& contact : contacts) {
1358 auto state_a = balls_[contact.a_index].sim.state();
1359 auto state_b = balls_[contact.b_index].sim.state();
1360
1361 Vector3 relative_velocity = state_b.velocity_mps - state_a.velocity_mps;
1362 const double normal_speed = relative_velocity.dot(contact.normal);
1363 const double position_bias =
1364 std::max(0.0, contact.penetration - field_.baumgarte_slop_m) *
1365 std::clamp(field_.baumgarte_beta, 0.0, 1.0) * inv_dt;
1366
1367 if (normal_speed < 0.0 || position_bias > 0.0) {
1368 const double inv_mass_sum = contact.inv_mass_a + contact.inv_mass_b;
1369 if (inv_mass_sum <= 1e-9) {
1370 continue;
1371 }
1372
1373 const double desired_normal_impulse =
1374 -((1.0 + contact.restitution) * normal_speed + position_bias) /
1375 inv_mass_sum;
1376 const double old_normal_accum = contact.normal_impulse_accum;
1377 contact.normal_impulse_accum = std::max(
1378 0.0, contact.normal_impulse_accum + desired_normal_impulse);
1379 const double applied_normal_impulse =
1380 contact.normal_impulse_accum - old_normal_accum;
1381 const Vector3 normal_impulse_vec = contact.normal * applied_normal_impulse;
1382
1383 state_a.velocity_mps -= normal_impulse_vec * contact.inv_mass_a;
1384 state_b.velocity_mps += normal_impulse_vec * contact.inv_mass_b;
1385
1386 relative_velocity = state_b.velocity_mps - state_a.velocity_mps;
1387 const Vector3 tangential_velocity =
1388 relative_velocity - contact.normal *
1389 relative_velocity.dot(contact.normal);
1390 const double tangential_speed = tangential_velocity.norm();
1391 if (tangential_speed > 1e-9 && contact.friction > 0.0) {
1392 const Vector3 tangent = tangential_velocity / tangential_speed;
1393 const double desired_tangent_impulse =
1394 -tangential_speed / inv_mass_sum;
1395 const double max_friction_impulse =
1396 contact.friction * contact.normal_impulse_accum;
1397 const double old_tangent_accum = contact.tangent_impulse_accum;
1398 contact.tangent_impulse_accum = std::clamp(
1399 contact.tangent_impulse_accum + desired_tangent_impulse,
1400 -max_friction_impulse, max_friction_impulse);
1401 const double applied_tangent_impulse =
1402 contact.tangent_impulse_accum - old_tangent_accum;
1403 const Vector3 friction_impulse = tangent * applied_tangent_impulse;
1404
1405 state_a.velocity_mps += friction_impulse * contact.inv_mass_a;
1406 state_b.velocity_mps -= friction_impulse * contact.inv_mass_b;
1407
1408 const double spin_transfer_gain =
1409 std::max(0.0, field_.ball_ball_spin_transfer_gain);
1410 if (spin_transfer_gain > 0.0) {
1411 const double inertia_a =
1412 std::max(1e-9, 0.4 * contact.mass_a * contact.radius_a * contact.radius_a);
1413 const double inertia_b =
1414 std::max(1e-9, 0.4 * contact.mass_b * contact.radius_b * contact.radius_b);
1415 state_a.spin_radps +=
1416 contact.normal.cross(friction_impulse) *
1417 (spin_transfer_gain / inertia_a) * contact.radius_a;
1418 state_b.spin_radps +=
1419 (-contact.normal).cross(-friction_impulse) *
1420 (spin_transfer_gain / inertia_b) * contact.radius_b;
1421 }
1422 }
1423
1424 balls_[contact.a_index].sim.setState(state_a);
1425 balls_[contact.b_index].sim.setState(state_b);
1426 }
1427 }
1428 }

1588 inverse.rotate(state.position_m - boundary.position_m);
1589
1590 const Vector3 closest(std::clamp(local.x, -boundary.half_extents_m.x,
1591 boundary.half_extents_m.x),
1592 std::clamp(local.y, -boundary.half_extents_m.y,
1593 boundary.half_extents_m.y),
1594 std::clamp(local.z, -boundary.half_extents_m.z,
1595 boundary.half_extents_m.z));
1596
1597 Vector3 delta = local - closest;
1598 double distance = delta.norm();
1599 Vector3 local_normal =
1600 distance > 1e-9 ? delta / distance : Vector3::unitZ();
1601
1602 if (distance >= ball_radius) {
1603 return;
1604 }
1605
1606 if (distance <= 1e-9) {
1607 const double pen_x = boundary.half_extents_m.x - std::abs(local.x);
1608 const double pen_y = boundary.half_extents_m.y - std::abs(local.y);
1609 const double pen_z = boundary.half_extents_m.z - std::abs(local.z);
1610
1611 if (pen_x <= pen_y && pen_x <= pen_z) {
1612 local_normal = Vector3(local.x >= 0.0 ? 1.0 : -1.0, 0.0, 0.0);
1613 distance = pen_x;
1614 } else if (pen_y <= pen_z) {
1615 local_normal = Vector3(0.0, local.y >= 0.0 ? 1.0 : -1.0, 0.0);
1616 distance = pen_y;
1617 } else {
1618 local_normal = Vector3(0.0, 0.0, local.z >= 0.0 ? 1.0 : -1.0);
1619 distance = pen_z;
1620 }
1621 }
1622
1623 const Vector3 world_normal =
1624 boundary.orientation.rotate(local_normal).normalized();
1625 state.position_m += world_normal *
1626 std::max(0.0, ball_radius - distance - field_.baumgarte_slop_m);
1627 const double impact_speed = std::max(0.0, -state.velocity_mps.dot(world_normal));
1628 applyContactImpulse(state.velocity_mps, world_normal,
1629 scaledWallRestitution(boundary.restitution,
1630 impact_speed),
1631 boundary.friction_coefficient);
