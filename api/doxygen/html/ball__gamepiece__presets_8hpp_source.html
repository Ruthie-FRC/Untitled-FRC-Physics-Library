122inline void configureSeason2026Field(BallGamepieceSim& sim) {
123 sim.setFieldConfig(season2026FieldConfig());
124 sim.fieldElements().clear();
125 sim.goals().clear();
126
127 EnvironmentalBoundary net{};
128 net.type = BoundaryType::kBox;
129 net.user_id = 2026;
130 net.position_m = Vector3(7.0, 4.1, 1.7);
131 net.half_extents_m = Vector3(0.4, 0.4, 0.5);
132 net.is_active = true;
133 sim.addFieldElement(net);
134
135 BallGamepieceSim::GoalZone hub_goal{};
136 hub_goal.shape = BallGamepieceSim::GoalZone::Shape::kBox;
137 hub_goal.center_m = net.position_m;
138 hub_goal.half_extents_m = Vector3(0.35, 0.35, 0.45);
139 hub_goal.accepted_type = BallGamepieceSim::GamePieceType::kBall;
140 hub_goal.require_positive_vertical_velocity = false;
141 sim.addGoalZone(hub_goal);
142}
