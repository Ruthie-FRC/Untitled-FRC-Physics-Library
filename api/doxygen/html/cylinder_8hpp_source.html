60inline Vector3 Cylinder::CalculateLocalInertia(double mass) const {
61 double r2 = m_radius * m_radius;
62 double h2 = 4.0 * m_halfHeight * m_halfHeight;
63 double ixz = (1.0 / 12.0) * mass * (3.0 * r2 + h2);
